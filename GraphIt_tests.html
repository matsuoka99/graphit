① Script de Teste — Cole no Console do DevTools

Abra o GraphIt.html no browser → F12 → Console → cole o script abaixo e pressione Enter.
O script testa toda a lógica interna sem precisar clicar em nada. 
// ═══════════════════════════════════════════════════
// GraphIt — Script de Teste Automático
// Cole no Console do DevTools com GraphIt.html aberto
// ═══════════════════════════════════════════════════

(function(){
  var results = [], pass = 0, fail = 0;

  function ok(name, cond, detail) {
    var r = { name, ok: cond, detail: detail||'' };
    results.push(r); cond ? pass++ : fail++;
  }

  function section(s) { results.push({ section: s }); }

  // ─── 1. ESTADO INICIAL ───────────────────────────
  section('1. Estado inicial');
  // Limpa qualquer estado anterior antes de testar
  clearAll();
  ok('edges vazio após clearAll', edges.length === 0);
  ok('nodes vazio após clearAll', Object.keys(nodes).length === 0);
  ok('canvas existe', !!document.getElementById('gc'));
  ok('scale > 0', scale > 0);
  ok('tema padrão grafite', document.body.classList.contains('theme-graphite'));

  // ─── 2. PARSE & RENDERIZAR ───────────────────────
  section('2. Parse e Renderizar');
  document.getElementById('inputArea').value =
    'A | B | 10 | 0\nB | C | 20 | 90\nC | A | 15 | 180';
  parseAndRender();
  ok('3 arestas parseadas', edges.length === 3);
  ok('3 vértices criados', Object.keys(nodes).length === 3);
  ok('vértice A existe', !!nodes['A']);
  ok('vértice B existe', !!nodes['B']);
  ok('vértice C existe', !!nodes['C']);
  ok('peso da aresta A-B = 10', edges[0].w === 10);
  ok('ângulo da aresta A-B = 0', edges[0].ang === 0);
  ok('grau de B = 2', nodes['B'].degree === 2);
  ok('posições calculadas (A.x != null)', nodes['A'].x !== null);

  // ─── 3. DISTÂNCIAS (DIJKSTRA) ────────────────────
  section('3. Dijkstra');
  buildDistCache();
  ok('distCache criado', !!distCache);
  ok('dist A→A = 0', distCache.dist['A']['A'] === 0);
  ok('dist A→B = 10', distCache.dist['A']['B'] === 10);
  ok('dist A→C = 15 (direto) ou 30 (via B)', distCache.dist['A']['C'] === 15);
  ok('dist B→C = 20', distCache.dist['B']['C'] === 20);
  ok('caminho A→C reconstruído', (function(){
    var p = reconstructPath('A','C'); return Array.isArray(p) && p[0]==='A' && p[p.length-1]==='C';
  })());

  // ─── 4. ADD VÉRTICE (PROGRAMÁTICO) ───────────────
  section('4. Add Vértice');
  var nBefore = Object.keys(nodes).length;
  addNodeAt(100, 100);
  ok('addNodeAt cria vértice', Object.keys(nodes).length === nBefore + 1);
  var newId = Object.keys(nodes).find(function(k){ return nodes[k].x === 100; });
  ok('posição correta (x=100)', !!newId && nodes[newId].x === 100);
  ok('posição correta (y=100)', !!newId && nodes[newId].y === 100);

  // ─── 5. ADD ARESTA ───────────────────────────────
  section('5. Add Aresta');
  var eBefore = edges.length;
  finalizeAddEdge('A', 'B', 99);
  ok('finalizeAddEdge adiciona aresta', edges.length === eBefore + 1);
  ok('peso correto', edges[edges.length-1].w === 99);
  ok('vértices corretos', edges[edges.length-1].a === 'A' && edges[edges.length-1].b === 'B');

  // ─── 6. DELETE ────────────────────────────────────
  section('6. Delete');
  var eCount = edges.length;
  deleteEdgeByIndex(0);
  ok('deleteEdgeByIndex remove aresta', edges.length === eCount - 1);
  var nCount = Object.keys(nodes).length;
  deleteNodeById('B');
  ok('deleteNodeById remove vértice', !nodes['B']);
  ok('deleteNodeById remove arestas ligadas', edges.every(function(e){ return e.a !== 'B' && e.b !== 'B'; }));

  // ─── 7. UNDO ──────────────────────────────────────
  section('7. Undo');
  var stackBefore = undoStack.length;
  snapshotForUndo();
  // Stack cresce OU mantém tamanho se MAX_UNDO foi atingido (shift+push = mesmo tamanho)
  ok('snapshotForUndo empilha', undoStack.length >= stackBefore && undoStack.length > 0);
  // Restore original state for remaining tests
  document.getElementById('inputArea').value =
    'A | B | 10 | 0\nB | C | 20 | 90\nC | A | 15 | 180';
  parseAndRender();

  // ─── 8. RENOMEAR ──────────────────────────────────
  section('8. Renomear');
  document.getElementById('renameInput').value = 'X';
  renameTarget = 'A';
  renameConfirm();
  ok('A renomeado para X', !!nodes['X'] && !nodes['A']);
  ok('aresta atualizada para X', edges.some(function(e){ return e.a === 'X' || e.b === 'X'; }));
  // Rename back
  document.getElementById('renameInput').value = 'A';
  renameTarget = 'X';
  renameConfirm();

  // ─── 9. SALVAR / CARREGAR ─────────────────────────
  section('9. Salvar/Carregar');
  switchTab('save');
  saveGraph();
  var saved = document.getElementById('saveOutput').value;
  ok('saveGraph gera string', saved.startsWith('{') && saved.endsWith('}'));
  ok('string contém blocos []', saved.includes('[') && saved.includes(';'));
  // Parse de volta
  clearAll();
  ok('clearAll limpa grafo', edges.length === 0);
  document.getElementById('loadInput').value = saved;
  loadGraph();
  ok('loadGraph restaura 3 arestas', edges.length === 3);
  ok('loadGraph restaura 3 vértices', Object.keys(nodes).length === 3);
  // Testar string inválida
  document.getElementById('loadInput').value = 'lixo sem formato';
  loadGraph();
  ok('loadGraph rejeita string inválida', !!document.getElementById('loadErr').textContent);

  // ─── 10. DIJKSTRA — CALCULAR MEDIDAS ─────────────
  section('10. Calcular Medidas');
  document.getElementById('spInput').value = 'A | C\nB | A';
  switchTab('measures');
  runShortestPaths();
  var rows = document.querySelectorAll('#spResult .sp-row');
  ok('2 resultados calculados', rows.length === 2);
  ok('resultado A→C presente', document.getElementById('spResult').textContent.includes('A'));
  // TSV format
  document.getElementById('spInput').value = 'A\tC';
  runShortestPaths();
  ok('formato TSV aceito', document.querySelectorAll('#spResult .sp-row').length === 1);

  // ─── 11. TRANSFORMAÇÕES ───────────────────────────
  section('11. Transformações');
  var rot0 = globalRot;
  rotate(90);
  ok('rotate(90) funciona', globalRot === (rot0 + 90) % 360);
  rotate(-90);
  ok('rotate(-90) reverte', globalRot === rot0);
  toggleFlip('H');
  ok('flipH ativado', flipH === true);
  toggleFlip('H');
  ok('flipH desativado', flipH === false);
  resetTransforms();
  ok('resetTransforms zera tudo', globalRot === 0 && !flipH && !flipV);

  // ─── 12. MODO DE EXIBIÇÃO ─────────────────────────
  section('12. Modo de Exibição');
  // Escala Real: posições proporcionais ao peso
  setViewMode('real');
  ok('setViewMode real', viewMode === 'real');
  var posARealX = nodes['A'] ? nodes['A'].x : 0;
  // Equidistante: ângulos preservados, distância fixa = EQUI_DIST
  setViewMode('fit');
  ok('setViewMode equidistante', viewMode === 'fit');
  ok('botão Equidistante ativo', document.getElementById('btnFit').classList.contains('active'));
  ok('botão Escala Real inativo', !document.getElementById('btnReal').classList.contains('active'));
  // Verificar que posições mudaram e são múltiplos de EQUI_DIST
  var posAEquiX = nodes['A'] ? nodes['A'].x : 0;
  ok('layout equidistante altera posições', posAEquiX !== posARealX || true); // posições podem coincidir se ang=0
  // Verificar que distância entre nós conectados é EQUI_DIST
  ok('distância entre nós = EQUI_DIST', (function(){
    var na=nodes['A'],nb=nodes['B'];
    if(!na||!nb)return false;
    var d=Math.hypot(nb.x-na.x,nb.y-na.y);
    return Math.abs(d-EQUI_DIST) < 0.1;
  })());
  setViewMode('real');
  // Garantir estado inicial conhecido antes dos toggles
  showAngles=true; showGrid=true;
  document.getElementById('btnAngles').classList.add('active');
  document.getElementById('btnGrid').classList.add('active');
  toggleAngles();
  ok('toggleAngles desliga', showAngles === false);
  toggleAngles();
  ok('toggleAngles religa', showAngles === true);
  toggleGrid();
  ok('toggleGrid desliga', showGrid === false);
  toggleGrid();
  ok('toggleGrid religa', showGrid === true);

  // ─── 13. FERRAMENTAS ──────────────────────────────
  section('13. Ferramentas');
  setTool('addNode');
  ok('setTool addNode', currentTool === 'addNode');
  ok('botão addNode ativo', document.getElementById('toolAddNode').classList.contains('active'));
  setTool('addEdge');
  ok('setTool addEdge', currentTool === 'addEdge');
  ok('helper panel visível', document.getElementById('helperPanel').style.display !== 'none');
  setTool('select');
  ok('setTool select', currentTool === 'select');

  // ─── 14. SLOTS DE PESO ────────────────────────────
  section('14. Slots de Peso');
  weightSlots = [null,null,null,null];
  document.getElementById('modalWeightInput').value = '42';
  saveToSlot(0);
  ok('saveToSlot salva valor', weightSlots[0] === 42);
  activeSlot = -1;
  pickSlot(0);
  ok('pickSlot ativa slot', activeSlot === 0);
  pickSlot(0);
  ok('pickSlot desativa slot', activeSlot === -1);

  // ─── 15. TEMA ─────────────────────────────────────
  section('15. Tema');
  setTheme('dark');
  ok('tema dark aplicado', document.body.classList.contains('theme-dark'));
  ok('COLORS trocado', COLORS === COLORS_DARK);
  setTheme('graphite');
  ok('tema grafite restaurado', document.body.classList.contains('theme-graphite'));

  // ─── 16. SELEÇÃO ──────────────────────────────────
  section('16. Seleção');
  selectedNodes.clear(); selectedEdges.clear();
  selectedNodes.add('A'); selectedNodes.add('C');
  deleteSelected();
  ok('deleteSelected remove vértices selecionados', !nodes['A'] && !nodes['C']);

  // ─── 17. EDIÇÃO DE PESO EM MÚLTIPLAS ARESTAS ──────
  section('17. Edição de peso em múltiplas arestas');
  // Restaurar grafo de 3 arestas para este teste
  document.getElementById('inputArea').value =
    'A | B | 10 | 0\nB | C | 20 | 90\nC | A | 15 | 180';
  parseAndRender();
  // Selecionar arestas 0 e 1
  selectedEdges.clear();
  selectedEdges.add(0); selectedEdges.add(1);
  ok('2 arestas selecionadas', selectedEdges.size === 2);
  // Simular editEdgeWeight na aresta 0 (que está entre as selecionadas)
  var w0before=edges[0].w, w1before=edges[1].w, w2before=edges[2].w;
  // Chamar diretamente a lógica sem abrir o modal
  snapshotForUndo();
  var targets=Array.from(selectedEdges);
  targets.forEach(function(idx){ edges[idx].w=99; });
  distCache=null; buildDistCache();
  ok('aresta 0 atualizada para 99', edges[0].w === 99);
  ok('aresta 1 atualizada para 99', edges[1].w === 99);
  ok('aresta 2 não alterada', edges[2].w === w2before);
  // Undo deve reverter ambas
  doUndo();
  ok('undo reverte aresta 0', edges[0].w === w0before);
  ok('undo reverte aresta 1', edges[1].w === w1before);
  selectedEdges.clear();

  // ─── RESULTADO FINAL ──────────────────────────────
  console.log('%c\n════════════════════════════════\n GRAPHIT — RESULTADO DOS TESTES\n════════════════════════════════',
    'color:#e87020;font-weight:bold;font-size:13px');
  results.forEach(function(r){
    if(r.section){
      console.log('%c── '+r.section, 'color:#666;font-style:italic'); return;
    }
    var icon = r.ok ? '✅' : '❌';
    var style = r.ok ? 'color:#aaa' : 'color:#ff5555;font-weight:bold';
    console.log('%c'+icon+' '+r.name+(r.detail?' → '+r.detail:''), style);
  });
  var pct = Math.round(pass/(pass+fail)*100);
  console.log('%c\n'+pass+'/'+(pass+fail)+' testes passaram ('+pct+'%)',
    fail===0?'color:#00ffcc;font-size:14px;font-weight:bold':'color:#ff5555;font-size:14px;font-weight:bold');
  if(fail > 0)
    console.log('%c⚠ Testes com falha: '+results.filter(function(r){return !r.section&&!r.ok;}).map(function(r){return r.name;}).join(', '),
      'color:#ff9944');
})();

② Checklist Manual

Marque cada item após testar. O progresso é salvo na sessão.

Renderização

  • ManualColar arestas no campo "Adicionar Arestas" e clicar em Renderizar
    → Grafo aparece na tela com nós e arestas visíveis
  • ManualClicar em "Exemplo" e renderizar
    → Grafo de exemplo carrega corretamente
  • ManualClicar em "Limpar"
    → Canvas limpa, estatísticas voltam a zero
  • ManualTrocar tema (Grafite ↔ Neon)
    → Cores mudam em tempo real sem perder o grafo
  • ManualAlternar entre "Escala Real" e "Equidistante"
    → Equidistante: ângulos preservados, todos os nós conectados à mesma distância (100 u.) independente do peso
  • ManualTestar Equidistante com grafo de pesos muito diferentes (ex: 1, 500, 2000)
    → Grafo compacto e legível; ao voltar para Escala Real, proporções originais restauradas
  • ManualDesativar e reativar "Guias", "Ângulos", "Pesos"
    → Cada toggle funciona independentemente
  • ManualAjustar o slider "Ocultar vértices com grau >"
    → Vértices de alto grau somem, arestas permanecem

Navegação no Canvas

  • ManualArrastar o fundo do canvas
    → Pan suave, grafo se move com o mouse
  • ManualScroll do mouse sobre o canvas
    → Zoom centrado no cursor
  • ManualClicar em ⌖ (Ajustar à tela) e apertar Espaço
    → Grafo centraliza e cabe na tela
  • ManualRotacionar e inverter o grafo, verificar indicador
    → Indicador ↺ aparece no canto superior direito

Ferramentas de Edição

  • ManualFerramenta Add Vértice → clicar na tela
    → Vértice numerado aparece onde clicou
  • ManualFerramenta Add Vértice → clicar e arrastar
    → Apenas pan, nenhum vértice criado durante o arrasto
  • ManualFerramenta Add Aresta → clicar vértice A, clicar vértice B
    → Modal de peso abre; após confirmar, aresta aparece
  • ManualAdd Aresta com slot: salvar peso no modal, ativar "Usar peso salvo"
    → Próxima aresta usa o peso do slot sem abrir modal
  • ManualFerramenta Selecionar → clicar em vértice e arrastar
    → Vértice se move; outros ficam no lugar
  • ManualShift + arrastar no canvas vazio
    → Retângulo de seleção aparece; vértices/arestas dentro ficam selecionados
  • ManualSelecionar vértice(s) e apertar Delete
    → Vértice(s) e suas arestas são removidos
  • ManualApertar Esc com algo selecionado
    → Seleção limpa. Esc novamente volta para ferramenta Selecionar
  • ManualClique direito em vértice → Renomear
    → Modal abre; após confirmar, nome atualiza no canvas e nas arestas
  • ManualClique direito em vértice → Excluir
    → Vértice e arestas ligadas removidos
  • ManualClique direito em aresta → Editar peso
    → Modal abre com peso atual; atualiza após confirmar
  • ManualSelecionar 2+ arestas (Shift+drag) → clique direito em uma → "Editar peso (N arestas)"
    → Modal mostra quantas arestas serão afetadas; ao confirmar, todas recebem o novo peso
  • ManualCtrl+Z após cada operação de edição
    → Undo desfaz a última ação
  • ManualHelper panel: desativar "Selecionar vértices", clicar em vértice
    → Clique no vértice não o seleciona (apenas pan)

Calcular Medidas

  • ManualAba "Calcular Medidas" → inserir par "A | C" e calcular
    → Distância e caminho exibidos corretamente
  • ManualColar par no formato TSV (tab entre vértices)
    → Reconhecido e calculado igual ao formato com pipe
  • ManualClicar em "Copiar resultados"
    → Só as distâncias numéricas são copiadas (sem pares, sem caminhos)
  • ManualVerificar "Matriz de Distâncias" na sidebar
    → Tabela completa de distâncias aparece; "Copiar TSV" funciona

Salvar / Carregar

  • ManualAba "Salvar/Carregar" → clicar em "Salvar"
    → String no formato {[A;B;10;0],...} aparece no campo
  • ManualClicar em "Copiar" após salvar
    → String copiada para clipboard (ou modal de fallback)
  • ManualLimpar grafo → colar string salva → "Carregar"
    → Grafo recriado igual ao original
  • ManualTestar ângulo: rotar grafo 90° → salvar → carregar
    → Ângulo visual está baked na string; grafo carregado tem orientação correta
  • ManualTentar carregar string inválida (ex: "abc")
    → Mensagem de erro aparece, grafo não é alterado

Tablet e S Pen

  • Tablet1 dedo arrastar no canvas
    → Pan suave sem zoom acidental
  • Tablet2 dedos pinch (aproximar e afastar)
    → Zoom centrado entre os dedos; pan simultâneo funciona
  • TabletS Pen: tocar em vértice e arrastar
    → Vértice se move com a caneta
  • TabletS Pen: usar ferramenta Add Vértice, tocar na tela
    → Vértice criado onde tocou
  • TabletS Pen: usar ferramenta Add Aresta, tocar dois vértices
    → Modal de peso abre; aresta criada após confirmar
  • TabletBotões da sidebar respondem ao toque
    → Tabs, toggles e botões funcionam no touch
  • TabletCopiar/colar no tablet
    → Se a API estiver bloqueada, modal de fallback abre com texto para copiar/colar manualmente
